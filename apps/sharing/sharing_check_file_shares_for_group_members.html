type id=user ${sharing_userA2}
type id=password ${sharing_user_password}

waitForText css=#header #expandDisplayName ${sharing_userA2}
waitForElementPresent css=#fileList tr[data-file="Shared"]

assertVisible css=#fileList tr[data-file="sharing file1.txt"]
mouseOver css=#fileList tr[data-file="sharing file1.txt"]

click css=#fileList tr[data-file="sharing file1.txt"] .fileactions a[data-action="Share"]
assertElementPresent css=#dropdown #shareWith[disabled="disabled"]
